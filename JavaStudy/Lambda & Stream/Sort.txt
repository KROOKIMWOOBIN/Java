import java.util.stream.*;
import java.util.*;

public class Main
{
	public static void main(String[] args) {
		
		List<Integer> list = Arrays.asList(3, 1, 4, 5, 2);
		List<Integer> sortedList = list.stream()
		                .sorted()
		                .collect(Collectors.toList());
		System.out.print(sortedList);
		
	}
}
